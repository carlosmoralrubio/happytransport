# 1. Install Firebase CLI (one time)
npm install -g firebase-tools
firebase login

# 2. Configure your environment
cd frontend
cp .env.example .env
# Edit .env → set VITE_API_BASE_URL and VITE_API_KEY

# 3. Configure and deploy the frontend (defaults to happytransport-logistics)
#    The script assumes a hosting site named 'happytransport-logistics' exists
#    in your Firebase project.  Create it once with:
#      firebase hosting:sites:create happytransport-logistics --project <your-project-id>
#    After that the deploy script will always publish to
#    https://happytransport-logistics.web.app.
chmod +x deploy-frontend.sh
./deploy-frontend.sh
#    you can export PROJECT_ID or HOSTING_SITE to override the defaults if
#    necessary (e.g. for testing)
